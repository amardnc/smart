Funktionsweise

Zwei Sensoren, zwei Aufgaben

Der erste Ultraschallsensor sitzt oben am Eimer und erkennt eine Hand, die sich nähert — daraufhin zieht der Servomotor den Deckel hoch. Der zweite Sensor misst von innen den Abstand zum Müll und rechnet ihn in einen Füllstand in Prozent um, der laufend auf dem LCD Display angezeigt wird. Überschreitet der Wert 80 %, piept der Buzzer als Erinnerung, den Eimer zu leeren.

Quellcode

Der Arduino-Sketch

Der vollständige C++ Code, der Sensoren, Servo, LCD Display und Buzzer steuert. Das komplette Projekt inklusive Schaltplan liegt im Repository.

smart_muelleimer.ino 
#include <Wire.h>
#include <LiquidCrystal_I2C.h>
#include <Servo.h>


LiquidCrystal_I2C lcd(0x27, 16, 2);

// ================= SERVO =================
Servo myServo;
const int servoPin = 9;

// ================= ULTRASCHALL 1 (Füllstand) =================
const int trig1 = 6;
const int echo1 = 4;
float kartonHoehe = 45.0;

// ================= ULTRASCHALL 2 (Bewegung) =================
const int trig2 = 8;
const int echo2 = 10;

// ================= BUZZER =================
const int buzzerPin = 7;

// ================= Abstand messen =================
float getDistance(int trigPin, int echoPin) {
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);

  long duration = pulseIn(echoPin, HIGH, 30000);

  if (duration == 0) return -1;

  return duration * 0.0343 / 2.0;
}

bool buzzerAlreadyTriggered = false;

void setup() {

  Serial.begin(9600);

  // Servo
  myServo.attach(servoPin);
  myServo.write(60);

  // Sensor 1
  pinMode(trig1, OUTPUT);
  pinMode(echo1, INPUT);

  // Sensor 2
  pinMode(trig2, OUTPUT);
  pinMode(echo2, INPUT);

  // LCD
  lcd.init();
  lcd.backlight();

  // Buzzer
  pinMode(buzzerPin, OUTPUT);
  digitalWrite(buzzerPin, LOW);
}

void loop() {

  // ================= FÜLLSTAND =================
  float d1 = getDistance(trig1, echo1);
  delay(60);

  float percent = 0;

  if (d1 > 0) {
    percent = ((kartonHoehe - d1) / kartonHoehe) * 100.0;
  }

  percent = constrain(percent, 0, 100);

  // Buzzer bei 80 %
  if (percent >= 80 && !buzzerAlreadyTriggered) {
    digitalWrite(buzzerPin, HIGH);
    delay(200);
    digitalWrite(buzzerPin, LOW);
    buzzerAlreadyTriggered = true;
  }

  if (percent < 80) {
    buzzerAlreadyTriggered = false;
  }

  // LCD-Anzeige
  lcd.setCursor(0, 0);

  if (percent >= 90) {
    lcd.print("Behaelter voll");
    lcd.setCursor(0, 1);
    lcd.print("Kein Oeffnen! ");
  } else {
    lcd.print("Fuellstand:   ");
    lcd.setCursor(0, 1);
    lcd.print(percent, 0);
    lcd.print("%     ");
  }

  // ================= BEWEGUNG =================
  float d2 = getDistance(trig2, echo2);
  delay(60);

  Serial.print("Motion: ");
  Serial.println(d2);

  // Nur öffnen, wenn der Füllstand unter 90 % liegt
  if (d2 > 0 && d2 < 20 && percent < 90) {

    // langsam auf
    for (int pos = 60; pos <= 110; pos++) {
      myServo.write(pos);
      delay(25);
    }

    delay(3000);

    // langsam zu
    for (int pos = 110; pos >= 60; pos--) {
      myServo.write(pos);
      delay(25);
    }
  }

  delay(300);
}
